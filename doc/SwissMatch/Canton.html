def initialize(license_tag, name, name_de, name_fr, name_it, name_rt)
  @license_tag  = license_tag
  @name         = name
  @names        = {
    :de => name_de,
    :fr => name_fr,
    :it => name_it,
    :rt => name_rt,
  }
end
